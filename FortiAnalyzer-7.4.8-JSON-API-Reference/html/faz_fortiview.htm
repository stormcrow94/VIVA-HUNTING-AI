Start FortiView Request

Run a FortiView new task.

Request:
{
    "params": [
        {
            "url": "/fortiview/adom/<adom-name>/<view-name>/run",
            "apiver": 3,
            "device": [
                {"devname": "string"},
                {"devid": "string"}
            ],
            "filter": "string",
            "sort-by": [
                {"field": "string", "order": "asc|desc"}
            ],
            "limit": number,
            "offset": number,
            "count-total": boolean,
            "time-range": {"start": "string", "end": "string"}
        }
    ],
    "jsonrpc": "2.0",
    "session": "...",
    "id": "..."
